PASUL 16 — COMENZI AI (PROPHET) — MESAJ SIMPLU + CANTITATE INTELIGENTA
================================================================================

UNDE: API/Chatbox.cs — BuildOrderProposalResponseAsync
      LangChain/stock_prediction.py + ai_server.py (cantitate Prophet)
      wwwroot/js/chatbox.js (buton Confirmă)

FLUX COMENZI AI:
1. Alege produsul cu cel mai mic stoc sub 20 buc. (din PostgreSQL).
2. Incarca istoric vanzari ultimele 14 zile (grupat pe zile, in memorie).
3. Calculeaza cantitatea:
   a) Daca serverul Python raspunde (GET /health + POST /chat):
      -> Prophet via stock_prediction.get_order_proposal() (~52 buc. la stoc 10).
   b) Altfel fallback .NET CalculeazaCantitateReaprovizionare():
      -> aceeasi formula (stoc tinta 30 zile, vanzari estimate, stoc critic).
4. Raspuns JSON catre frontend:
   - answer: „**Propunere:** {Nume} — stoc actual **{X}** bucăți”
     (FARA ROP, FARA „Prophet”, FARA vanzari/zi in mesajul principal)
   - action: { type: execute_order, product, quantity }

BUTON CONFIRMĂ (chatbox.js):
- „Comandăm {quantity} buc de {product}?”
- La confirmare: POST /api/chat/execute-order -> creste stoc + email furnizor.

FORMULA CANTITATE (stock_prediction.py / mirror C#):
- vanzari_zilnice_est: Prophet sau media istoric; default 2.0 buc./zi.
- stoc_siguranta = ceil(vanzari * lead_time * 0.2), lead_time = 3 zile.
- stoc_tinta = ceil(vanzari * 30 zile + stoc_siguranta).
- cantitate = stoc_tinta - stoc_curent; minim 10 buc. la stoc critic (<= 15).

DE CE APAREA 20 BUc. (temporar):
- Versiune simplificata C#: max(10, 30 - stoc) -> 20 la stoc 10.
- Inlocuita cu formula Prophet + apel optional Python.

ERORI REZOLVATE:
- EF: DateTime.ToString in LINQ (nu se traduce in SQL) -> grupare in memorie.
- Build CS1061: days.Count pe object -> tipizare corecta Dictionary.
- Eroare Comenzi AI la pregatire date -> mutare completa in .NET + Prophet optional.

    public static async Task SeedAsync(ApplicationDbContext context, ILogger logger)
    {
        var catalog = BuildCatalog();

        logger.LogInformation("Seeding catalog: {Categories} categorii, {Total} produse.",
            catalog.Count, catalog.Sum(c => c.Value.Count));

        await context.OrderItems.ExecuteDeleteAsync();
        await context.Favorites.ExecuteDeleteAsync();
        await context.Products.ExecuteDeleteAsync();

        await context.Database.ExecuteSqlRawAsync(
            @"ALTER TABLE ""Products"" ALTER COLUMN ""Id"" RESTART WITH 1");

        var allProducts = new List<Product>();
        var baseDate = DateTime.UtcNow.AddDays(-30);

        foreach (var (category, items) in catalog)
        {
            for (var i = 0; i < items.Count; i++)
            {
                var (name, desc, price, qty) = items[i];
                allProducts.Add(new Product
                {
                    Name = name,
                    Category = category,
                    Description = desc,
                    Price = price,
                    Quantity = qty,
                    Imagine = "default.jpg",
                    AddedDate = baseDate.AddDays(i)
                });
            }
        }

        context.Products.AddRange(allProducts);
        await context.SaveChangesAsync();

        logger.LogInformation("Catalog reseed complet: {Count} produse.", allProducts.Count);
    }

using Microsoft.EntityFrameworkCore;
using ProiectWeb.Models;

namespace ProiectWeb.Data;

public static class DatabaseMaintenance
{
    /// <summary>
    /// Șterge toate comenzile și liniile asociate.
    /// </summary>
    public static async Task ClearAllOrdersAsync(ApplicationDbContext db)
    {
        await db.OrderItems.ExecuteDeleteAsync();
        await db.Orders.ExecuteDeleteAsync();
    }

    /// <summary>
    /// Ajustează prețurile produselor: 0–500 RON, majoritatea sub 100 RON.
    /// Setează AcquisitionCost la ~65% din preț (marjă minimă pentru AI).
    /// </summary>
    public static async Task AdjustProductPricesAsync(ApplicationDbContext db)
    {
        var products = await db.Products.ToListAsync();
        if (products.Count == 0)
            return;

        var premiumCategories = new HashSet<string>(StringComparer.OrdinalIgnoreCase)
        {
            "Calculatoare", "Telefoane", "Laptopuri", "Electronice", "TV & Audio"
        };

        foreach (var p in products)
        {
            decimal price;
            if (premiumCategories.Contains(p.Category))
            {
                price = 150 + (p.Id % 35) * 10; // 150–490
            }
            else if (p.Category.Contains("Bauturi", StringComparison.OrdinalIgnoreCase)
                     || p.Category.Contains("Aliment", StringComparison.OrdinalIgnoreCase)
                     || p.Category.Contains("Fructe", StringComparison.OrdinalIgnoreCase)
                     || p.Category.Contains("Legume", StringComparison.OrdinalIgnoreCase))
            {
                price = 5 + (p.Id % 9) * 5; // 5–45
            }
            else
            {
                price = 15 + (p.Id % 17) * 5; // 15–95
            }

            if (price > 500) price = 500;
            if (price < 1) price = 1;

            p.Price = price;
            p.AcquisitionCost = Math.Round(price * 0.65m, 2);
        }

        await db.SaveChangesAsync();
    }
}

namespace ProiectWeb.Data;

/// <summary>
/// Populează istoricul de comenzi pentru testare (recomandări, statistici, notificări).
/// Rulează o singură dată: dotnet run -- --seed-orders
/// </summary>
public static class OrderHistorySeeder
{
    private const string Email1 = "raresmarian3344@gmail.com";
    private const string Email2 = "raresmarian604@gmail.com";
    private const string DefaultPassword = "TestUser123!";
    private const int MinAppearancesPerProduct = 3;

    public static async Task SeedAsync(IServiceProvider services, bool force = false)
    {
        var db = services.GetRequiredService<ApplicationDbContext>();
        var userManager = services.GetRequiredService<UserManager<ApplicationUser>>();

        var u1 = await userManager.FindByEmailAsync(Email1);
        var u2 = await userManager.FindByEmailAsync(Email2);
        var userIds = new[] { u1?.Id, u2?.Id }.Where(id => id != null).Cast<string>().ToList();

        if (userIds.Count > 0 && !force)
        {
            var existingForUsers = await db.Orders.CountAsync(o => userIds.Contains(o.UserId));
            if (existingForUsers > 0)
            {
                Console.WriteLine($"[Seed] Există deja {existingForUsers} comenzi pentru acești utilizatori.");
                Console.WriteLine("[Seed] Folosește --seed-orders --force pentru a adăuga din nou.");
                return;
            }
        }

        var products = await db.Products.OrderBy(p => p.Id).ToListAsync();
        if (products.Count == 0)
        {
            Console.WriteLine("[Seed] Nu există produse în baza de date.");
            return;
        }

        var user1 = await EnsureUserAsync(userManager, Email1, "Rares");
        var user2 = await EnsureUserAsync(userManager, Email2, "Marian");

        var byCategory = products.GroupBy(p => p.Category).ToDictionary(g => g.Key, g => g.ToList());
        var categories = byCategory.Keys.ToList();

        var stock = products.ToDictionary(p => p.Id, p => p.Quantity);
        var plannedOrders = BuildOrderPlan(products, byCategory, categories, user1.Id, user2.Id);

        plannedOrders = plannedOrders.OrderBy(o => o.Date).ToList();

        int created = 0;
        foreach (var plan in plannedOrders)
        {
            var validItems = new List<(Product Product, int Quantity)>();
            foreach (var (productId, preferredQty) in plan.Items)
            {
                var product = products.FirstOrDefault(p => p.Id == productId);
                if (product == null) continue;

                var qty = ResolveQuantity(stock, productId, preferredQty);
                if (qty <= 0) continue;

                validItems.Add((product, qty));
            }

            if (validItems.Count == 0) continue;

            var order = new Order
            {
                UserId = plan.UserId,
                OrderDate = plan.Date,
                IsPaid = true,
                StripeSessionId = null,
            };

            foreach (var (product, qty) in validItems)
            {
                order.Items.Add(new OrderItem { ProductId = product.Id, Quantity = qty });
            }

            db.Orders.Add(order);
            await db.SaveChangesAsync();

            foreach (var (product, qty) in validItems)
            {
                var qtyBefore = stock[product.Id];
                stock[product.Id] = Math.Max(0, qtyBefore - qty);
                product.Quantity = stock[product.Id];

                db.Notifications.Add(new Notification
                {
                    Message = $"Stoc scăzut: «{product.Name}» {qtyBefore} → {product.Quantity} buc. (comandă #{order.Id}).",
                    DateCreated = plan.Date,
                });
            }

            await db.SaveChangesAsync();
            created++;
        }

        Console.WriteLine($"[Seed] {created} comenzi create pentru {Email1} și {Email2}.");
        Console.WriteLine($"[Seed] Produse acoperite: {products.Count}, perioadă ~4 luni până acum.");
    }

    private static int ResolveQuantity(Dictionary<int, int> stock, int productId, int preferred)
    {
        if (!stock.TryGetValue(productId, out var available) || available <= 0)
            return 0;

        var qty = Math.Clamp(preferred, 1, 4);
        if (available < qty) qty = available;
        return qty;
    }

    private static List<OrderPlan> BuildOrderPlan(
        List<Product> products,
        Dictionary<string, List<Product>> byCategory,
        List<string> categories,
        string user1Id,
        string user2Id)
    {
        var plans = new List<OrderPlan>();
        var now = DateTime.UtcNow;

        var dayOffsets = new[]
        {
            118, 112, 105, 98, 92, 85, 78, 72, 65, 58,
            52, 45, 38, 32, 28, 24, 21, 18, 14, 11,
            8, 6, 4, 3, 2, 1
        };

        var userIds = new[] { user1Id, user2Id };

        var pool = new List<int>();
        foreach (var p in products)
            for (int k = 0; k < MinAppearancesPerProduct; k++)
                pool.Add(p.Id);
        ShuffleDeterministic(pool, seed: 42);

        int poolIdx = 0;
        for (int i = 0; i < dayOffsets.Length; i++)
        {
            var daysAgo = dayOffsets[i];
            var hour = 9 + (i * 3) % 11;
            var minute = (i * 17) % 60;
            var date = now.Date.AddDays(-daysAgo).AddHours(hour).AddMinutes(minute);
            var userId = userIds[i % 2];

            int basketSize = daysAgo > 60 ? 6 : daysAgo > 30 ? 5 : 4;
            var items = TakeBasketFromPool(pool, ref poolIdx, basketSize, products, byCategory, categories, i);

            plans.Add(new OrderPlan(date, userId, items));
        }

        while (poolIdx < pool.Count)
        {
            var extraDay = 90 - (plans.Count % 80);
            if (extraDay < 1) extraDay = 1;
            var date = now.Date.AddDays(-extraDay).AddHours(14 + plans.Count % 6);
            var userId = userIds[plans.Count % 2];
            var items = TakeBasketFromPool(pool, ref poolIdx, 5, products, byCategory, categories, plans.Count);
            plans.Add(new OrderPlan(date, userId, items));
        }

        return plans;
    }

    private static List<(int ProductId, int Quantity)> TakeBasketFromPool(
        List<int> pool,
        ref int poolIdx,
        int basketSize,
        List<Product> products,
        Dictionary<string, List<Product>> byCategory,
        List<string> categories,
        int orderIndex)
    {
        var items = new List<(int ProductId, int Quantity)>();
        var used = new HashSet<int>();

        while (items.Count < basketSize && poolIdx < pool.Count)
        {
            var pid = pool[poolIdx++];
            if (!used.Add(pid)) continue;
            items.Add((pid, PreferredQty(orderIndex, pid)));
        }

        if (items.Count >= 2 && categories.Count > 1)
        {
            var productMap = products.ToDictionary(p => p.Id);
            var catsInBasket = items
                .Select(i => productMap.GetValueOrDefault(i.ProductId)?.Category)
                .Where(c => c != null)
                .Distinct()
                .ToList();

            if (catsInBasket.Count == 1)
            {
                var otherCat = categories.First(c => c != catsInBasket[0]);
                var cross = byCategory[otherCat]
                    .OrderBy(p => (p.Id + orderIndex) % byCategory[otherCat].Count)
                    .FirstOrDefault(p => !used.Contains(p.Id));

                if (cross != null)
                {
                    used.Add(cross.Id);
                    items.Add((cross.Id, 1));
                }
            }
        }

        return items;
    }

    private static int PreferredQty(int orderIndex, int productId)
    {
        var roll = (orderIndex * 7 + productId * 13) % 10;
        return roll switch
        {
            >= 8 => 3,
            >= 4 => 2,
            _ => 1,
        };
    }

    private static void ShuffleDeterministic(List<int> list, int seed)
    {
        var rng = new Random(seed);
        for (int i = list.Count - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            (list[i], list[j]) = (list[j], list[i]);
        }
    }

    private static async Task<ApplicationUser> EnsureUserAsync(
        UserManager<ApplicationUser> userManager,
        string email,
        string nickname)
    {
        var user = await userManager.FindByEmailAsync(email);
        if (user != null) return user;

        user = new ApplicationUser
        {
            UserName = email,
            Email = email,
            EmailConfirmed = true,
            Nickname = nickname,
        };

        var result = await userManager.CreateAsync(user, DefaultPassword);
        if (!result.Succeeded)
            throw new InvalidOperationException(
                $"Nu am putut crea utilizatorul {email}: {string.Join(", ", result.Errors.Select(e => e.Description))}");

        await userManager.AddToRoleAsync(user, "User");
        return user;
    }

    private sealed record OrderPlan(DateTime Date, string UserId, List<(int ProductId, int Quantity)> Items);
}
